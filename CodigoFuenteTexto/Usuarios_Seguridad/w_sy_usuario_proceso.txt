public subroutine wf_aplicar_filtro ();string ls_filtro_planilla, ls_filtro_proceso, ls_filter

ls_filtro_planilla = trim(upper(sle_filtro_planilla.text))
ls_filtro_proceso = trim(upper(sle_filtro_proceso.text))

ls_filter = ""

// Filtrar por código en lugar de descripción
if ls_filtro_planilla <> "" then
	ls_filter = "upper(tipoplanilla) like '%" + ls_filtro_planilla + "%'"
end if

if ls_filtro_proceso <> "" then
	if ls_filter <> "" then
		ls_filter += " and "
	end if
	ls_filter += "upper(tipoproceso) like '%" + ls_filtro_proceso + "%'"
end if

// Aplicar o limpiar filtro
if ls_filter = "" or isnull(ls_filter) then
	dw_planillaproceso.setfilter("")
	dw_planillaproceso.filter()
else
	dw_planillaproceso.setfilter(ls_filter)
	dw_planillaproceso.filter()
end if
end subroutine

	ls_column = LEFT(dwo.Name, LEN(String(dwo.Name)) - 2) 
	ls_old_sort = Describe("Datawindow.Table.sort") 
	IF ls_column = LEFT(ls_old_sort, LEN(ls_old_sort) - 2) THEN 
		lc_sort = RIGHT(ls_old_sort, 1)  
		IF lc_sort = 'A' THEN 
			lc_sort = 'D' 
			ELSE 
			lc_sort = 'A' 
		END IF 
		SetSort(ls_column+" "+lc_sort) 
	ELSE 
		SetSort(ls_column+" A") 
	END IF 
	Sort() 
END IF
